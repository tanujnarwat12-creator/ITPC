
#include <stdio.h>

int main() {

   int num, sum = 0, rev = 0, digit;

   printf("Enter a 3-digit number: ");

   scanf("%d", &num);

 

   int temp = num;

   while(temp != 0) {

       digit = temp % 10;

       sum += digit;

       rev = rev * 10 + digit;

       temp /= 10;

   }

 

   printf("Sum of digits = %d\n", sum);

   printf("Reverse = %d\n", rev);

 

   return 0;

}
